using System;

namespace Lab_1_Q2
{
    class Program
    {
        static void Main(string[] args)
        {
            int sum = 0;
            Console.WriteLine("The first 10 natural numbers are:");

            for (int i = 1; i <= 10; i++)
            {
                Console.Write(i + " ");
                sum += i;
            }

            Console.WriteLine($"\nThe sum of the first 10 natural numbers is: {sum}");
        }
    }
}

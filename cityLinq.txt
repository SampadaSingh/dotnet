
using System;
using System.Linq;

class Program
{
    static void Main()
    {
        string[] cities = { "ROME", "LONDON", "NAIROBI", "CALIFORNIA", "ZURICH", "NEWDELHI", "AMSTERDAM", "ABU DHABI", "PARIS" };

        var result = cities.Where(c => c.StartsWith("N") && c.EndsWith("I"));

        Console.WriteLine("Cities that start with 'N' and end with 'I':");

        foreach (var city in result)
            Console.WriteLine(city);
    }
}
